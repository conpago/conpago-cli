	/**
	 * Command definition. Implement it to add new command to Cli.

	interface ICommand {
		/**
		 * Prints help to presenter.
		 *
		 * @return void
		 */
		function printHelp();

		/**
		 * Returns description of command.
		 *
		 * @return string
		 */
		function getDescription();

		/**
		 * Runs command.
		 *
		 * @param string[] $args
		 *
		 * @return void
		 */
		function run(array $args);
	}
