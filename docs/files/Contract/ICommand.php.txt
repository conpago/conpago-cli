    interface ICommand
    {
        /**
         * Prints help to presenter.
         *
         * @return void
         */
        public function printHelp();

        /**
         * Returns description of command.
         *
         * @return string
         */
        public function getDescription();

        /**
         * Runs command.
         *
         * @param string[] $args
         *
         * @return void
         */
        public function run(array $args);
    }
